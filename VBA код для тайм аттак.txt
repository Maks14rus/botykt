    Dim seconds As Double

    If IsNumeric(timeValue.Value) Then

        ' »звлекаем доли секунд
        milliseconds = (timeValue.Value - Int(timeValue.Value)) * 24 * 60 * 60  ' ѕреобразуем остаток времени в миллисекунды
